### **创建**

``` c++
int create_BT(BTN_Ptr *btp)
{
    int temp_data = 0;
    std::cin >> temp_data;
    if (temp_data == 0)
    {
        *btp = NULL;
        printf("leaf\n");
    }
    else
    {
        if ( !(*btp = (BTN_Ptr)malloc( sizeof(BTN) ) ) ) 
        {
            printf("error : malloc error");
            return -1;
        }
        (*btp)->data = temp_data;
        create_BT(&((*btp)->LeftChild));
        create_BT(&((*btp)->RightChild));
    } 
    return 0;
}
```
